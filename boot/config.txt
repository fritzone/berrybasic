# BerryBasic - bare-metal Raspberry Pi 4 boot configuration.
#
# The Pi 4 boot ROM + on-board EEPROM load start4.elf/fixup4.dat from this FAT
# partition, which then load and run kernel8.img (our 64-bit bare-metal kernel).

# Run the kernel in 64-bit (AArch64) mode.
arm_64bit=1

# Our kernel image.
kernel=kernel8.img

# Load the Pi 4 device tree normally. We do NOT disable it (no "device_tree=")
# because on Pi 4 the firmware's 64-bit kernel handoff depends on a DTB being
# present - disabling it makes the firmware fail to start the kernel (splash,
# then nothing). Our bare-metal kernel simply ignores the DTB pointer the
# firmware passes in x0.

# --- HDMI: force a known-good, monitor-independent mode ---------------------
# We pin HDMI to exactly the resolution the kernel's framebuffer is compiled for,
# and drive it unconditionally regardless of what the monitor reports over EDID
# (some monitors/TVs don't advertise the mode, which would give a black "no
# signal" screen):
#   hdmi_force_hotplug=1  - always output, even if hotplug detect says "no TV"
#   hdmi_ignore_edid      - don't trust the monitor's mode list; use ours
#   hdmi_drive=2          - normal HDMI mode (with audio); use 1 for pure DVI
# The resolution block below is regenerated by tools/configure.sh, using a custom
# CVT timing (hdmi_mode=87) so any resolution works.
hdmi_force_hotplug=1
hdmi_ignore_edid=0xa5000080
hdmi_drive=2
config_hdmi_boost=5
# >>> berrybasic-resolution (generated by tools/configure.sh) >>>
hdmi_group=2
hdmi_mode=87
hdmi_cvt=1280 720 60
# <<< berrybasic-resolution <<<

# Serial console on GPIO14/15, 115200 8N1. Handy for headless bring-up.
enable_uart=1

# Show the rainbow splash on purpose: if you see it, HDMI + firmware output are
# working and any later black screen is our kernel's framebuffer, not the
# cable/monitor. (Set back to 1 once the screen is confirmed working.)
disable_splash=0

# We only need a framebuffer, so keep GPU memory small.
gpu_mem=64
